## C++ Code

```C++
#include <iostream>
using namespace std;

int main()
{
	int x_point[3];
	int y_point[3];

	for (int i = 0; i < 3; i++)
	{
		cin >> x_point[i] >> y_point[i];
	}

	int x_alone = 0;
	int y_alone = 0;

	if (x_point[0] == x_point[1])
		x_alone = x_point[2];
	else if (x_point[1] == x_point[2])
		x_alone = x_point[0];
	else if (x_point[0] == x_point[2])
		x_alone = x_point[1];

	if (y_point[0] == y_point[1])
		y_alone = y_point[2];
	else if (y_point[1] == y_point[2])
		y_alone = y_point[0];
	else if (y_point[0] == y_point[2])
		y_alone = y_point[1];

	cout << x_alone << ' ' << y_alone;
}
```
