## C++ Code

```C++
#include <iostream>
using namespace std;

int inhabitants(int k, int n)
{
	if (k == 1)
	{
		int result = 0;

		for (int i = 1; i <= n; i++)
		{
			result += i;
		}
		return result;
	}
	else
	{
		int result = 0;

		for (int i = 1; i <= n; i++)
		{
			result += inhabitants(k - 1, i);
		}
		return result;
	}
}


int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int T, k, n;
	cin >> T;

	for (int i = 0; i < T; i++)
	{
		cin >> k >> n;

		cout << inhabitants(k, n) << '\n';
	}
}
```
