## C++ Code

```C++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL); cout.tie(NULL);

	int a;
	int N;
	cin >> N;

	vector<int> time;

	for (int i = 0; i < N; i++)
	{
		cin >> a;
		time.push_back(a);
	}

	sort(time.begin(), time.end());

	int sum = 0;
	int total = 0;

	for (int i = 0; i < N; i++)
	{
		sum = 0;
		for (int j = 0; j <= i; j++)
		{
			sum += time[j];
		}
		total += sum;
	}

	cout << total;
}
```

## Python Code

```Python
import sys
input = sys.stdin.readline

n = int(input())

waiting_num = list(map(int, input().split()))

waiting_num.sort()
total_num = 0

for i in range(n):
    total_num += waiting_num[i] * (n - i)

print(total_num)
```
