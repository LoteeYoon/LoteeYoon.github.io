## C++ Code

```C++
#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int N;
	cin >> N;

	int p_number[1000];
	fill_n(p_number, 1000, 1);
	p_number[0] = 0, p_number[1] = 0;

	for (int i = 2; i < 1000; i++)
	{
		for (int j = 2 * i; j < 1000; j += i)
		{
			p_number[j] = 0;
		}
	}

	int cnt = 0;

	for (int i = 0; i < N; i++)
	{
		int isPrime;
		cin >> isPrime;

		if (p_number[isPrime] == 1)
			cnt += 1;
	}

	cout << cnt;
}
```
