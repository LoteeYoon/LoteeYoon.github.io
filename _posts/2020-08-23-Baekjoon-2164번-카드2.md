## C++ Code

```C++
#include <iostream>
#include <queue>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL); cout.tie(NULL);

	int N;
	cin >> N;

	queue<int> q;

	for (int i = 1; i <= N; i++)
	{
		q.push(i);
	}

	while (true)
	{
		int tmp = q.front();
		q.pop();

		if (q.empty() == true)
		{
			cout << tmp;
			break;
		}
		else
		{
			tmp = q.front();
			q.pop();
			q.push(tmp);
		}
	}
}
```
