## C++ Code

```C++
#include <iostream>
using namespace std;

long long memory[100] = { 0, 1, };

long long fibonacci(int n)
{
	for (int i = 2; i <= n; i++)
	{
		memory[i] = memory[i - 1] + memory[i - 2];
	}
	return memory[n];
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL); cout.tie(NULL);

	int N;
	cin >> N;

	cout << fibonacci(N);
}
```
