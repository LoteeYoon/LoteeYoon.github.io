## C++ Code

```C++
#include <iostream>
#include <vector>
#include <numeric>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	vector<int> Prime;
	int M, N;
	cin >> M >> N;

	int p_number[10000];
	fill_n(p_number, 10000, 1);
	p_number[0] = 0; p_number[1] = 0;

	for (int i = 2; i < 10000; i++)
	{
		for (int j = 2 * i; j < 10000; j += i)
		{
			p_number[j] = 0;
		}
	}

	for (int i = M; i <= N; i++)
	{
		if (p_number[i] == 1)
			Prime.push_back(i);
	}

	if (Prime.empty() == true)
		cout << -1 << '\n';
	else
	{
		cout << accumulate(Prime.begin(), Prime.end(), 0) << '\n';
		cout << Prime[0] << '\n';
	}
}
```
