## C++ Code

```C++
#include <iostream>
using namespace std;

int main()
{
	int x, y, w, h;
	cin >> x >> y >> w >> h;

	int a[4] = { w - x, h - y, x, y };
	int min = a[0];

	for (int i = 1; i < 4; i++)
	{
		if (a[i] < min)
			min = a[i];
	}

	cout << min;
}
```
