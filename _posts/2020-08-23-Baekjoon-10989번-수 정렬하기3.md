## C++ Code

```C++
#include <iostream>
using namespace std;

int count_arr[10000] = { 0 };

void count_sort(void)
{
	for (int i = 1; i <= 10000; i++)
	{
		for (int j = 1; j <= count_arr[i]; j++)
		{
			cout << i << '\n';
		}
	}
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL); cout.tie(NULL);

	int N;
	cin >> N;

	int number = 0;

	for (int i = 0; i < N; i++)
	{
		cin >> number;
		count_arr[number] += 1;
	}

	count_sort();
}
```
