## C++ Code

```C++
#include <iostream>
#include <string>
#include <vector>
using namespace std;

vector<string> v = { "c=", "c-", "dz=", "d-", "lj", "nj", "s=", "z=" };

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL); cout.tie(NULL);

	int idx = 0;
	string s;

	cin >> s;

	for (int i = 0; i < v.size(); i++)
	{
		while (1)
		{
			idx = s.find(v[i]);
			if (idx == string::npos)
				break;
			s.replace(idx, v[i].length(), "#");
		}

	}

	cout << s.length();
}
```
