## C++ Code

```C++
#include <iostream>
#include <string>
using namespace std;

int main()
{
	string upper = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
	string lower = "abcdefghijklmnopqrstuvwxyz";
	string s;
	cin >> s;
	int count[26] = { 0 };

	for (int i = 0; i < s.length(); i++)
	{
		if (upper.find(s[i]) != string::npos)
			count[upper.find(s[i])] += 1;
		else if (lower.find(s[i]) != string::npos)
			count[lower.find(s[i])] += 1;
	}

	int max = 0;
	int overlap = 0;
	int index;

	for (int j = 0; j < 26; j++)
	{
		if (max < count[j])
		{
			max = count[j];
			index = j;
		}
	}

	for (int k = 0; k < 26; k++)
	{
		if (max == count[k])
			overlap += 1;
	}

	if (overlap >= 2)
		cout << "?";
	else
		cout << (char)(index + 65);
}
```
