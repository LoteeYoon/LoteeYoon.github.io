## C++ Code

```C++
#include <iostream>
using namespace std;

int factorial(int n)
{
	int result = 1;

	for (int i = n; i >= 1; --i)
	{
		result *= i;
	}

	return result;
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL); cout.tie(NULL);

	int N, K;
	cin >> N >> K;

	int ans = 0;

	ans = factorial(N) / (factorial(K) * factorial(N - K));

	cout << ans;
}
```
