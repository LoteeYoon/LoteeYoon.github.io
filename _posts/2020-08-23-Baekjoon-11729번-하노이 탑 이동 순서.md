## C++ Code

```C++
#include <iostream>
using namespace std;


void Hanoi(int n, int from, int by, int to)
{
	if (n == 1)
	{
		cout << from << ' ' << to << '\n';
	}
	else
	{
		Hanoi(n - 1, from, to, by);
		cout << from << ' ' << to << '\n';
		Hanoi(n - 1, by, from, to);
	}
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int N;
	cin >> N;

	cout << (1 << N) - 1 << '\n';
	Hanoi(N, 1, 2, 3);
}
```
