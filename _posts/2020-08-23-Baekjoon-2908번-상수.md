## C++ Code

```C++
#include <iostream>
#include <string>
using namespace std;

int main()
{
	string A, B;
	cin >> A >> B;

	char temp;
	temp = A[0];
	A[0] = A[2];
	A[2] = temp;
	temp = B[0];
	B[0] = B[2];
	B[2] = temp;

	int i = stoi(A);
	int j = stoi(B);

	if (i > j)
		cout << i;
	else
		cout << j;
}
```
