## C++ Code

```C++
#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int N, M;
	cin >> N >> M;

	int card[100];
	fill_n(card, 100, 0);
	int sum = 0;
	int max = 0;

	for (int i = 0; i < N; i++)
	{
		cin >> card[i];
	}

	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < N; j++)
		{
			for (int k = 0; k < N; k++)
			{
				if (i != j && j != k && k != i)
				{
					sum = card[i] + card[j] + card[k];

					if (sum <= M && max < sum)
						max = sum;
				}
			}
		}
	}

	cout << max << '\n';
}
```
