## C++ Code

```C++
#include <iostream>
#include <string>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL); cout.tie(NULL);

	int N;
	cin >> N;

	int count = 0;
	int title = 665;
	string s;

	while (++title)
	{
		s = to_string(title);

		if (s.find("666") != -1)
			count++;

		if (count == N)
		{
			cout << title << '\n';
			break;
		}
	}
}
```
