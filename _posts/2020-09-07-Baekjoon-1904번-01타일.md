## C++ Code

```C++
#include <iostream>
#include <vector>
#define MOD 15746
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL); cout.tie(NULL);

	int N;
	cin >> N;

	vector<long long> dp = { 0, 1, 2 };

	for (int i = 3; i <= N; ++i)
	{
		int tmp;
		tmp = dp[i - 1] + dp[i - 2];
		dp.push_back(tmp % MOD);
	}

	cout << dp[N] % MOD;
}
```
