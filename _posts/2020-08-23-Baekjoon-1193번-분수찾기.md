## C++ Code

```C++
#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int x;
	cin >> x;

	int th = 0;

	while (x > 0)
	{
		++th;
		x -= th;
	}

	int denominator;
	int numerator;

	if (th % 2 != 0)
	{
		denominator = 1;
		numerator = th;

		for (int i = 0; i < x + th - 1; i++)
		{
			denominator++;
			numerator--;
		}
	}
	else
	{
		denominator = th;
		numerator = 1;

		for (int i = 0; i < x + th - 1; i++)
		{
			denominator--;
			numerator++;
		}
	}

	cout << numerator << '/' << denominator << '\n';
}
```
