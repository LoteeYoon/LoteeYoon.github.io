## C++ Code

```C++
#include <iostream>
#include <string>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL); cout.tie(NULL);

	int sec[26] = { 3, 3, 3, 4, 4, 4, 5, 5, 5, 6, 6, 6, 7, 7, 7, 8, 8, 8, 8, 9, 9, 9, 10, 10, 10, 10 };
	int time = 0;
	string s;

	cin >> s;

	for (int i = 0; i < s.length(); i++)
	{
		int idx = s[i] - 65;
		time += sec[idx];
	}

	cout << time << '\n';

}
```
