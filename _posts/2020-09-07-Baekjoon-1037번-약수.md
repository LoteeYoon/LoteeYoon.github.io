## C++ Code

```C++
#include <iostream>
#include <algorithm>
#include <cmath>
#include <vector>
using namespace std;

vector<int> divisor;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL); cout.tie(NULL);

	int N;
	cin >> N;

	for (int i = 0; i < N; ++i)
	{
		int a;
		cin >> a;
		divisor.push_back(a);
	}

	sort(divisor.begin(), divisor.end());

	if (divisor.size() % 2 == 0)
	{
		int result = 0;
		result = divisor[(divisor.size() / 2) - 1] * divisor[divisor.size() / 2];
		cout << result;
	}
	else
	{
		int result = 0;
		result = pow(divisor[divisor.size() / 2], 2);
		cout << result;
	}
}
```
