## C++ Code

```C++
#include <iostream>
#include <vector>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL); cout.tie(NULL);

	int T, N;
	cin >> T;

	vector<long long> dp = { 0, 1, 1, 1 };

	for (int j = 4; j <= 100; ++j)
	{
		long long tmp;
		tmp = dp[j - 3] + dp[j - 2];
		dp.push_back(tmp);
	}

	for (int i = 0; i < T; ++i)
	{
		cin >> N;
		cout << dp[N] << '\n';
	}
}
```
