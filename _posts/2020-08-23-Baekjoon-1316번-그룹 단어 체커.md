## C++ Code

```C++
#include <iostream>
#include <string>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL); cout.tie(NULL);

	int N, j;
	int cnt = 0;
	int arr[26];
	string s;
	cin >> N;

	for (int i = 0; i < N; i++)
	{
		cin >> s;
		fill_n(arr, 26, 0);

		arr[s[0] - 97] = 1;

		for (j = 1; j < s.length(); j++)
		{
			if (s[j - 1] != s[j])
			{
				int idx = s[j] - 97;
				if (arr[idx] == 0)
					arr[idx]++;
				else
					break;
			}
		}

		if (j == s.length())
			cnt++;
	}

	cout << cnt;
}
```
